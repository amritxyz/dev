% Introduction
\begin{frame}{Introduction}
	\begin{enumerate}
		\item<1-> "CASE tools" stands for \textcolor{alert}{Computer-Aided Software Engineering} tools. \pause
		\item<2-> These are software applications that support software development and engineering processes.
	\end{enumerate}
\end{frame}


% What are CASE tools
\begin{frame}{What are \textbf{CASE Tools}?}
	\begin{enumerate}
		\item<1-> \textbf{CASE Tools} help system analysts to: \pause
			\begin{enumerate}
				\item Develop information systems.
				\item Maintain information systems. \pause
			\end{enumerate}
		\item<2-> Main goal: Improve \textcolor{alert}{productivity, quality, consistency} and
			\textcolor{alert}{automate} the software development lifecycle.
	\end{enumerate}
\end{frame}

\begin{frame}{Types of CASE Tools}
	\begin{enumerate}
		\item \textbf{Upper CASE Tools} \pause
		\item \textbf{Lower CASE Tools}
	\end{enumerate}
\end{frame}

% Types of CASE Tools
\begin{frame}{Types of CASE Tools}
	% \begin{columns}
	% \column{0.5\textwidth}
	\begin{enumerate}
		\item<1->\textbf{Upper CASE Tools} \pause
			\begin{enumerate}
				\item<1-> Focus on planning and analysis phases.
				\item<2-> Examples: Requirements analysis, UML, ERD tools, Gantt charts. \pause
			\end{enumerate}

			% \column{0.5\textwidth}
		\item<2->\textbf{Lower CASE Tools} \pause
			\begin{enumerate}
				\item<3-> Focus on implementation, testing, and maintenance.
				\item<4-> Examples: Code generators, debuggers, version control.
			\end{enumerate}
	\end{enumerate}
	% \end{columns}
\end{frame}

\begin{frame}{Integrated CASE Tools}
	\begin{enumerate}
		\item<1-> Support the \textit{entire software development lifecycle}. \pause
		\item<2-> Examples: 
			\begin{enumerate}
				\item<3-> \only<3,4,5,6,7,8,9,10>{Rational Rose} \pause
					\only<4>{(Owned By \textcolor{alert}{IBM}. Forward/Reverse engineering.)} \pause
				\item<5-> \only<5,6,7,8,9,10>Enterprise Architect \pause
					\only<6>{(Owned By \textcolor{alert}{SPARX}. Code Eng, Testing tools.)} \pause
				\item<7-> \only<7,8,9,10>Visual Paradigm \pause
					\only<8>{(\textcolor{alert}{SDLC} . Plan, Design, Build and Documentation.)} \pause

			\end{enumerate}
	\end{enumerate}
\end{frame}

% Advantages
\begin{frame}{Advantages of CASE Tools}
	\begin{enumerate}
		\item<1-> Increases productivity by automating routine tasks. \pause
		\item<2-> Improves software quality through standardization. \pause
		\item<3-> Enhances documentation and communication. \pause
		\item<4-> Reduces development time and cost. \pause
		\item<5-> Facilitates early detection of errors.
	\end{enumerate}
\end{frame}

% Disadvantages
\begin{frame}{Disadvantages of CASE Tools}
	\begin{enumerate}
		\item<1-> High initial cost of acquisition and training. \pause
		\item<2-> Steep learning curve for some tools. \pause
		\item<3-> Resistance to change from traditional methods. \pause
		\item<4-> May not fit all project types or sizes. \pause
		\item<5-> Dependency on tool support and updates.
	\end{enumerate}
\end{frame}
